```typescript
// apps/server/src/routes/agents.ts
import { Hono } from 'hono'
import { zValidator } from '@hono/zod-validator'
import { z } from 'zod'
import type { Adapter } from '../adapter'

export function agentRoutes(adapter: Adapter) {
  const app = new Hono()

  app.get('/', async (c) => {
    const agents = await adapter.listAgents()
    return c.json(agents)
  })

  app.get('/:id', async (c) => {
    const id = c.req.param('id')
    const agent = await adapter.getAgent(id)
    return c.json(agent)
  })

  app.post('/', zValidator('json', z.object({
    name: z.string(),
    model: z.string().optional(),
  })), async (c) => {
    const body = c.req.valid('json')
    const agent = await adapter.createAgent(body)
    return c.json(agent, 201)
  })

  app.delete('/:id', async (c) => {
    const id = c.req.param('id')
    await adapter.deleteAgent(id)
    return c.json({ ok: true })
  })

  return app
}
```

### 3.2 SSE дәӢд»¶жҺЁйҖҒ

```typescript
// apps/server/src/routes/sse.ts
import { Hono } from 'hono'
import { streamSSE } from 'hono/streaming'
import type { Adapter } from '../adapter'

export function sseRoutes(adapter: Adapter) {
  const app = new Hono()

  app.get('/stream', (c) => {
    return streamSSE(c, async (stream) => {
      const handler = (event: { type: string; payload: unknown }) => {
        stream.writeSSE({
          event: event.type,
          data: JSON.stringify(event.payload),
        })
      }

      adapter.on('event', handler)
      stream.onAbort(() => adapter.off('event', handler))
    })
  })

  return app
}
```

### 3.3 еә”з”Ёз»„иЈ…

```typescript
// apps/server/src/app.ts
import { Hono } from 'hono'
import { cors } from 'hono/cors'
import { logger } from 'hono/logger'
import { agentRoutes } from './routes/agents'
import { chatRoutes } from './routes/chat'
import { sseRoutes } from './routes/sse'
import { skillRoutes } from './routes/skills'
import { sessionRoutes } from './routes/sessions'
import { workspaceRoutes } from './routes/workspace'
import { executionRoutes } from './routes/executions'
import { projectRoutes } from './routes/projects'
import { modelRoutes } from './routes/models'
import { cronRoutes } from './routes/cron'
import { settingRoutes } from './routes/settings'
import type { Adapter } from './adapter'

export function createApp(adapter: Adapter) {
  const app = new Hono()

  // дёӯй—ҙд»¶
  app.use('*', logger())
  app.use('/api/*', cors())

  // Gateway иҝһжҺҘзҠ¶жҖҒжЈҖжҹҘ
  app.use('/api/*', async (c, next) => {
    if (c.req.path === '/api/status') return next()
    if (!adapter.isConnected()) {
      return c.json({ error: 'Gateway disconnected' }, 503)
    }
    return next()
  })

  // и·Ҝз”ұжҢӮиҪҪ
  app.route('/api/agents', agentRoutes(adapter))
  app.route('/api/chat', chatRoutes(adapter))
  app.route('/api/skills', skillRoutes(adapter))
  app.route('/api/sessions', sessionRoutes(adapter))
  app.route('/api/workspace', workspaceRoutes(adapter))
  app.route('/api/executions', executionRoutes(adapter))
  app.route('/api/projects', projectRoutes(adapter))
  app.route('/api/models', modelRoutes(adapter))
  app.route('/api/cron', cronRoutes(adapter))
  app.route('/api/settings', settingRoutes(adapter))
  app.route('/api/events', sseRoutes(adapter))

  // зі»з»ҹзҠ¶жҖҒ
  app.get('/api/status', (c) => {
    return c.json({
      gateway: adapter.isConnected() ? 'connected' : 'disconnected',
      uptime: process.uptime(),
    })
  })

  return app
}
```
